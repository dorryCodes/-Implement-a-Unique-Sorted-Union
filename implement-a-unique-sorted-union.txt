function uniteUnique(...args){
args=args.join();//to join my arrays
let arg=new Set(args) //to make the values in the array unique.
let arr=[...arg].filter(x=>x !==',').map(y=>parseInt(y)); //[...arg] this spreading and encapsulation is to make it an array again
return arr
}
console.log(uniteUnique([1,2,4],[2,3,4]))
